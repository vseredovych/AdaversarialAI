	\section{–ú–æ–¥–∏—Ñ—ñ–∫–æ–≤–∞–Ω–∞ –ª–æ–≥—ñ—Å—Ç–∏—á–Ω–∞ —Ä–µ–≥—Ä–µ—Å—ñ—è}
	–ó–∞–¥–∞—á–∞ –ø—Ä–∏–∫–ª–∞–¥—É $x \in R^{n_x}$, –∑–Ω–∞–π—Ç–∏ $\hat{y}=P(y = 1 \mid x), 0 \leq \hat{y} \leq 1$
	–®—É–∫–∞—Ç–∏–º–µ–º–æ —É –≤–∏–≥–ª—è–¥—ñ $\hat{y} = \sigma (\omega\tran x + b)$, –¥–µ
	\begin{itemize}
		\item –ø–∞—Ä–∞–º–µ—Ç—Ä–∏ $\omega \in R^{n_x}, b \in R$ - –Ω–µ–≤—ñ–¥–æ–º—ñ, –ø–æ—Ç—Ä—ñ–±–Ω–æ –∑–Ω–∞–π—Ç–∏ –æ–ø—Ç–∏–º–∞–ª—å–Ω—ñ –¥–ª—è –¥–ª—è –¥–∞–Ω–æ—ó –∑–∞–¥–∞—á—ñ
		\item $\sigma(z) = \frac{1}{1+ e^{-z}}$ - —Å–∏–≥–º–æ—ó–¥
	\end{itemize}
	–î–ª—è –∫–æ–∂–Ω–æ–≥–æ –ø—Ä–∏–∫–ª–∞–¥—É –∑ —Ç—Ä–µ–Ω—É–≤–∞–ª—å–Ω–æ–≥–æ –¥–∞—Ç–∞—Å–µ—Ç—É –ø–æ—Ç—Ä—ñ–±–Ω–æ –æ–±—á–∏—Å–ª–∏—Ç–∏:
	\begin{itemize}
		\item $ z\ith = \omega\tran x\ith + b $
		\item $ y\ith = \sigma\ith (z\ith) $
	\end{itemize}
	–¥–µ $\sigma(z\ith) = \frac{1}{1 + e^{-z\ith}}$, —Ç–∞–∫ —â–æ–± $\hat{y}\ith \approx y\ith $ \newline
	–î–ª—è –∫–æ–∂–Ω–æ–≥–æ –ø—Ä–∏–∫–ª–∞–¥—É –≤–∏–∑–Ω–∞—á–µ–Ω–∞ —Ñ—É–Ω–∫—Ü—ñ—è –≤—Ç—Ä–∞—Ç–∏:
	$$L(\hat{y}\ith, y\ith) = -y\ith \log(\hat{y}\ith) - (1 - y\ith) \log(1 - \hat{y}\ith)$$
	–ù–∞ –≤—Å—ñ—Ö –ø—Ä–∏–∫–ª–∞–¥–∞—Ö –æ–±—á–∏—Å–ª—é—î–º–æ —à—Ç—Ä–∞—Ñ–Ω—É —Ñ—É–Ω–∫—Ü—ñ—é —è–∫:
	$$J(w, b) = \frac{1}{m} \sum_{i=1}^{m} L(\hat{y}\ith, y\ith)$$

	\section{FGSM} 
	–ü–µ—Ä—à–∏–º –º–µ—Ç–æ–¥–æ–º –∞—Ç–∞–∫–∏ —è–∫–∏–π –±—É–¥–µ —Ä–æ–∑–≥–ª—è–¥–∞—Ç–∏—Å—å —î –º–µ—Ç–æ–¥—É —à–≤–∏–¥–∫–æ–≥–æ –≥—Ä–∞–¥—ñ—î–Ω—Ç—É (Fast Gradient Sign Method).
	
	–û–¥–Ω–æ–∫—Ä–æ–∫–æ–≤–∏–π –º–µ—Ç–æ–¥ –≥—Ä–∞–¥—ã—ç–Ω—Ç–Ω–æ–≥–æ —Å—Ç—É—Å–∫—É, FGSM, —â–æ–± –∑–Ω–∞–π—Ç–∏ —Ç–∞–∫–∏–π adversarial –ø—Ä–∏–∫–ª–∞–¥ $x^{*}$ —è–∫–∏–π –º–∞–∫—Å–∏–º—ã–∑—É—é —Ñ—É–Ω–∫—Ü—ñ—é –≤—Ç—Ä–∞—Ç–∏ $J(x^{*}, y)$, –¥–µ J —î —Ñ—É–Ω–∫—Ü—ñ—î—é –∫—Ä–æ—Å—Å-–µ–Ω—Ç—Ä–æ–ø—ñ—ó.
	
	$ X^{*} = X + \epsilon \cdot sign(\Delta_x J(x, y))$, 
	
	–¥–µ $\Delta_x J(x, y)$
	
	
	–Ü–¥–µ—è –ø–æ–ª—è–≥–∞—î –≤ —Ç–æ–º—É ..=.
	
	\begin{align*}
		z_i = \omega\tran x\ith + b \quad
		\hat{y}_i = a_i = softmax(z_i) \quad
		softmax(z_i) = \frac{e^{z_i}}{\sum_{k=1}^{c} e_k^z} \quad
	\end{align*}
	\textbf{–ü–æ—Ö—ñ–¥–Ω–∞ softmax —Ñ—É–Ω–∫—Ü—ñ—ó} \newline
	–Ø–∫—â–æ $i = j$,
	\begin{align*}
	    \pdv{\hat{y_i}}{z_j} 
	    &=
	    \pdv{\frac{e^{z_i}}{\sum_{k=1}^{c} e_k^z}}{z_j} 
	    =
	    \frac{e^{z_i} \sum_{k=1}^{c} e^{z_k} - e^{z_j} e^{z_i}}{(\sum_{k=1}^{c}  e^{z_k})^2} 
	    = \\& =
	    \frac{e^{z_j}}{\sum_{k=1}^{c}  e^{z_k}} \times \frac{(\sum_{k=1}^{c} e^{z_k} - e^{z_j})}{\sum_{k=1}^{c}  e^{z_k}} 
	    = 
	    y_i(1-y_j)
	\end{align*}
	–Ø–∫—â–æ $i \neq j$,
	\begin{align*}
		\pdv{\hat{y_i}}{z_j}
		&=
		\pdv{\frac{e^{z_i}}{\sum_{k=1}^{c} e_k^z}}{z_j} 
		=
		\frac{0 - e^{z_j} e^{z_i}}{\sum_{k=1}^{c}  e^{z_k}} \times \frac{e^{z_i}}{\sum_{k=1}^{c} e^{z_k}} 
		= 
		-y_i y_j 
	\end{align*}
	
	\textbf{–§—É–Ω–∫—Ü—ñ—è –∫—Ä–æ—Å—Å-–µ–Ω—Ç—Ä–æ–ø—ñ—ó}
	\begin{align*}
		\xi(y, x) = - \sum_{i=1}^{c} y_i  \log (\hat{y_i})
	\end{align*}
	
	\textbf{–§—É–Ω–∫—Ü—ñ—è –∫—Ä–æ—Å—Å-–µ–Ω—Ç—Ä–æ–ø—ñ—ó}
	\begin{align*}
		\pdv{\xi}{z_i} 
		&= 
		- \sum_{j=1}^{c} \pdv{ y_j \log (\hat{y_j})}{z_i} 
		=
		- \sum_{j=1}^{c} y_j \pdv{ \log (\hat{y_j})}{z_i} 
		= \\&
		- \sum_{j=1}^{c} y_j \frac{1}{\hat{y_i}} \cdot \pdv{\hat{y_i}}{z_i} 
		= 
		- \frac{y_i}{\hat{y_i}} \pdv{\hat{y_i}}{z_i} - \sum_{j \neq i} \frac{y_j}{\hat{y_j}} \pdv{\hat{y_j}}{z_i} 
		= \\& =
		-\frac{y_i}{\hat{y_i}} \hat{y_i} (1 - \hat{y_i}) - \sum_{j \neq i} \frac{y_j}{\hat{y_j}} (-\hat{y_i} \hat{y_j}) 
		=
		-y_i + y_i \hat{y_i} + \sum_{j \neq i} y_j \hat{y_i}
		= \\& =
		-y_i + \sum_{j=1}^{c} y_j \hat{y_i} 
		=
		\hat{y_i} - y_i
	\end{align*}
	, i = 1, ..., C
	
	\textbf{–û–±—á–∏—Å–ª–µ–Ω–Ω—è —É—Ä–∞–¥—ñ—î–Ω—Ç—É "–º–∞—î –±—É—Ç–∏ –≤–∏—â–µ"}
	\begin{align*}
		\pdv{\xi}{x_i} 
		&=
		\pdv{\xi}{z_i} \pdv{z_i}{x_i} 
		=
		(\hat{y_i} - y_i) \pdv{(\omega\tran x_i+ b)}{x_i} 
		=
		(\hat{y_i} - y_i) x_i 
		=
		dz \cdot x
	\end{align*}
	
	\textbf{–û–±—á–∏—Å–ª–µ–Ω–Ω—è —É—Ä–∞–¥—ñ—î–Ω—Ç—É "–º–∞—î –±—É—Ç–∏ –≤–∏—â–µ"}
